Gramatica:
---------
<programa> ::= receita de <id> : ingredientes <ingredientes>
         procedimentos <procedimentos>
<ingredientes> ::= <ingrediente> . <ingredientes>
         | <ingrediente>
<ingrediente> ::= farinha
         | ovos
         | manteiga
         | leite
         | sal
         | acucar
         | canela
         | fermento
         | nescau
         | leitemoca
<procedimentos> ::= <procedimento> ; <procedimentos>
         | <procedimento>
<procedimento> ::= <colocar>
         | <bater>
         | <usar>
         | <assar>
         | <servir>
<colocar> ::= colocar <ingrediente> na tigela
         | colocar <ingrediente> na forma
         | colocar <ingrediente> no liquidificador
         | colocar <ingrediente> na batedeira
         | colocar <ingrediente> no forno
<bater> ::= bater <numero> minutos
<usar> ::= usar liquidificador
         | usar batedeira
         | usar forno
<assar> ::= assar por <numero> minutos
<servir> ::= servir
